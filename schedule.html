        const toYYYYMMDD = (date) => date.toISOString().split('T')[0];

       

        function getHierarchicalProjects() {

            const projectMap = new Map(projects.map(p => [p.id, { ...p, children: [] }]));

            const hierarchical = [];

            for (const project of projectMap.values()) {

                if (project.parentId && projectMap.has(project.parentId)) {

                    projectMap.get(project.parentId).children.push(project);

                } else {

                    hierarchical.push(project);

                }

            }

            const flatList = [];

            function flatten(project, depth) {

                flatList.push({ ...project, depth });

                project.children.sort((a,b) => a.id - b.id).forEach(child => flatten(child, depth + 1));

            }

            hierarchical.sort((a,b) => a.id - b.id).forEach(p => flatten(p, 0));

            return flatList;

        }

            else if (currentView === 'year') renderYearView(filteredProjects);

        function navigate(direction) {

            if (currentView === 'month') currentDate.setMonth(currentDate.getMonth() + direction);

            else if (currentView === 'week') currentDate.setDate(currentDate.getDate() + (7 * direction));

            else if (currentView === 'day') currentDate.setDate(currentDate.getDate() + direction);

            else if (currentView === 'year') currentDate.setFullYear(currentDate.getFullYear() + direction);

            render();

        }

        function saveProject(e) {

            e.preventDefault();

            const form = DOM['project-form'].elements;

            const id = parseInt(form['project-id'].value);

            const parentId = form['project-parent'].value ? parseInt(form['project-parent'].value) : null;

           

            // Save main project

            const projectData = {

                name: form['project-name'].value,

                status: form['project-status'].value,

                parentId: parentId,

            };

 

            if (id) { // Editing

                const index = projects.findIndex(p => p.id === id);

                projects[index] = { ...projects[index], ...projectData };

            } else { // Adding new

                const newId = projects.length > 0 ? Math.max(...projects.map(p => p.id)) + 1 : 1;

                projectData.id = newId;

                projects.push(projectData);

 

                // Add AI-generated subtasks if any

                if (tempSubtasks.length > 0) {

                    tempSubtasks.forEach(taskName => {

                        const subtaskId = projects.length > 0 ? Math.max(...projects.map(p => p.id)) + 1 : 1;

                        projects.push({

                            id: subtaskId,

                            name: taskName,

                            status: '規劃中',

                            parentId: newId

                        });

                    });

                }

            }

            closeModal(DOM['project-modal']);

            render();

        }

       

        function deleteProject(projectId) {

            const childrenIds = getHierarchicalProjects()

                .filter(p => p.parentId === projectId)

                .map(p => p.id);

 

            childrenIds.forEach(childId => deleteProject(childId)); // Recursive delete

 

            projects = projects.filter(p => p.id !== projectId);

            delete dailyTasks[projectId];

           

            closeModal(DOM['project-modal']);

            render();

        }

       

        async function handleAIBreakdown() {

            const promptText = DOM['ai-prompt'].value;

            if (!promptText) {

                alert('請先輸入專案目標！');

                return;

            }

 

            DOM['ai-btn-text'].classList.add('hidden');

            DOM['ai-spinner'].classList.remove('hidden');

            DOM['ai-breakdown-btn'].disabled = true;

 

            const chatHistory = [{ role: "user", parts: [{ text: `將以下專案目標拆解成 5 到 8 個主要的、可執行的子任務列表： "${promptText}"。請僅用繁體中文回答。` }] }];

            const payload = {

                contents: chatHistory,

                generationConfig: {

                    responseMimeType: "application/json",

                    responseSchema: {

                        type: "OBJECT",

                        properties: {

                            "subtasks": {

                                "type": "ARRAY",

                                "items": { "type": "STRING" }

                            }

                        }

                    }

                }

            };

            const apiKey = "";

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

 

            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

 

                if (!response.ok) {

                    throw new Error(`API 請求失敗，狀態碼：${response.status}`);

                }

 

                const result = await response.json();

               

                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const jsonText = result.candidates[0].content.parts[0].text;

                    const parsedJson = JSON.parse(jsonText);

                   

                    if (parsedJson.subtasks && parsedJson.subtasks.length > 0) {

                        tempSubtasks = parsedJson.subtasks;

                        DOM['ai-results'].innerHTML = `

                            <p class="font-semibold mb-1">建議的子任務：</p>

                            <ul class="list-disc list-inside bg-white p-2 rounded">

                                ${tempSubtasks.map(task => `<li>${task}</li>`).join('')}

                            </ul>`;

                    } else {

                        DOM['ai-results'].textContent = 'AI 未能生成有效的子任務，請嘗試不同的提示。';

                    }

                } else {

                    DOM['ai-results'].textContent = '從 AI 收到的回應格式不符預期。';

                }

            } catch (error) {

                console.error('AI 任務拆解失敗:', error);

                DOM['ai-results'].textContent = `發生錯誤：${error.message}`;

            } finally {

                DOM['ai-btn-text'].classList.remove('hidden');

                DOM['ai-spinner'].classList.add('hidden');

                DOM['ai-breakdown-btn'].disabled = false;

            }

        }

 

        function openDailyTaskModal(projectId, date) {

            const project = projects.find(p => p.id == projectId);

            if (!project || project.parentId === null) {

                alert('請在子任務上新增工作項目。');

                return;

            }

            document.getElementById('daily-task-title').textContent = `${project.name} - ${date}`;

            document.getElementById('daily-task-project-id').value = projectId;

            document.getElementById('daily-task-date').value = date;

            document.getElementById('daily-task-content').value = (dailyTasks[projectId] && dailyTasks[projectId][date]) || '';

            DOM['daily-task-modal'].classList.remove('hidden');

        }

       

        function saveDailyTask(e) {

            e.preventDefault();

            const projectId = document.getElementById('daily-task-project-id').value;

            const date = document.getElementById('daily-task-date').value;

            const content = document.getElementById('daily-task-content').value;

            if (!dailyTasks[projectId]) dailyTasks[projectId] = {};

            if (content) {

                dailyTasks[projectId][date] = content;

            } else {

                delete dailyTasks[projectId][date];

            }

            closeModal(DOM['daily-task-modal']);

            render();

        }

       

        function populateParentDropdown(currentProjectId = null) {

            const dropdown = document.getElementById('project-parent');

            dropdown.innerHTML = '<option value="">無 (設為主專案)</option>';

            projects.forEach(p => {

                if (p.id === currentProjectId || p.parentId !== null) return;

                dropdown.add(new Option(p.name, p.id));

            });

        }

 

        function populateSelectors() {

            const startYear = 2025;

            for (let y = startYear; y < startYear + 11; y++) DOM['year-select'].add(new Option(`${y}年`, y));

            for (let m = 0; m < 12; m++) DOM['month-select'].add(new Option(`${m + 1}月`, m));

        }

       

        function exportToExcel() {

            const wb = XLSX.utils.book_new();

            const projectData = projects.map(p => ({'專案/任務ID': p.id, '名稱': p.name, '主專案ID': p.parentId, '狀態': p.status }));

            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(projectData), "專案總表");

            const dailyTaskData = [];

            for (const pid in dailyTasks) {

                for (const date in dailyTasks[pid]) {

                    const project = projects.find(p => p.id == pid);

                    dailyTaskData.push({ '任務ID': pid, '任務名稱': project ? project.name : 'N/A', '日期': date, '工作項目': dailyTasks[pid][date] });

                }

            }

            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(dailyTaskData), "每日工作項目");

            XLSX.writeFile(wb, `工作計畫表_${toYYYYMMDD(new Date())}.xlsx`);

        }

 

        populateSelectors();

        setupEventListeners();

        render();

    });
